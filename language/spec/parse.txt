NOTES:
  - types and values use different namespaces
  - a specific token shall be between ``
  - the special token empty shall be the absence of a token

file =
  | item-definition*

item-definition =
  | `func` definition

definition-start =
  | ident parameter-list? declared-type?
definition :=
  | definition-start block
  | definition-start `=` expression `;`

declared-type =
  | `:` type

parameter-list = 
  | `(` parameter-list-no-parens `)`
parameter-list-no-parens =
  | parameter `,`?
  | parameter `,` parameter-list-no-parens
parameter =
  | ident declared-type

type =
  | identifier

block =
  | `{` statement* expression `}`

statement =
  | `let` definition
  | expression `;`

expression =
  | literal
  | expression binop expression
  | `if` expression `{` expression `}`
    -- NOTE(ubsan): should just take a block
  | `-` expression
