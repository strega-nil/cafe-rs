NOTES:
  - types and values use different namespaces
  - a specific token shall be between ``
  - the special token empty shall be the absence of a token

file :=
  | item-definition*

item-definition :=
  | func-definition

func-definition := 
  | `func` ident parameter-list opt-return-type `::` block

parameter-list := 
  | `(` parameter-list-no-parens `)`
parameter-list-no-parens :=
  | parameter `,`?
  | parameter `,` parameter-list-no-parens
parameter :=
  | ident `:` type

opt-return-type :=
  | empty
  | `->` type

type :=
  | identifier

block :=
  | `{` statement* expression `}`

statement :=
  | expression `;`

expression :=
  | number-literal
  | expression binop expression
