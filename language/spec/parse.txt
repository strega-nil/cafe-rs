NOTES:
  - types and values use different namespaces
  - a specific token shall be between ``
  - the special token empty shall be the absence of a token

file :=
  item-definition*

item-definition :=
  | value-definition

value-definition :=
  identifier `::` value-definition-kind

value-definition-kind :=
  fn-definition

fn-definition := 
  | parameter-list opt-return-type block
parameter-list :=
  | parameter
  | parameter `,` parameter-list
  | parameter-list `,`
parameter :=
  | ident `:` type

opt-return-type :=
  | empty
  | `->` type

type :=
  | identifier

block :=
  `{` statement* expression `}`

statement :=
  | expression `;`

expression :=
  | number-literal
  | expression binop expression
