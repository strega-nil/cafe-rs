NOTES:
  - types and values use different namespaces
  - a specific token shall be between ``
  - the special token empty shall be the absence of a token
  - the definition syntax is not yet fully implemented in the
    compiler

# idea: maybe allow leaving off the `=` before the `{` of a block

file =
  | item-definition*

item-definition =
  | func-definition

func-definition = 
  | `func` definition

definition-start =
  | ident parameter-list? `:` type
definition :=
  # | definition-start block
  | definition-start `=` block
  | definition-start `=` expression `;`

parameter-list = 
  | `(` parameter-list-no-parens `)`
parameter-list-no-parens =
  | parameter `,`?
  | parameter `,` parameter-list-no-parens
parameter =
  | ident `:` type

type =
  | identifier

block =
  | `{` statement* expression `}`

statement =
  | `let` definition
  | expression `;`

expression =
  | number-literal
  | expression binop expression
